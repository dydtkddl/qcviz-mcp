```python
# -*- coding: utf-8 -*-
"""
PySCF 기반 IAO/IBO 계산 백엔드 구현.

Phase η 추가:
- X2C 스칼라 상대론 (compute_scf_relativistic)
- 란타나이드/악티나이드 MINAO 폴백 (_resolve_minao 확장)
- 적응적 SCF 수렴 에스컬레이션 (ConvergenceStrategy, compute_scf_adaptive)
"""
from __future__ import annotations

import logging
from pathlib import Path
from typing import Any

import numpy as np

from qcviz_mcp.backends.base import IAOResult, IBOResult, OrbitalBackend, SCFResult
from qcviz_mcp.backends.registry import registry

try:
    import pyscf
    from pyscf import gto, lo, scf
    from pyscf.tools import cubegen
    _HAS_PYSCF = True
except ImportError:
    _HAS_PYSCF = False

logger = logging.getLogger(__name__)

_SUPPORTED_METHODS = frozenset({"HF", "RHF", "UHF", "RKS", "UKS", "B3LYP", "PBE0"})

# ── Phase η: 란타나이드/악티나이드 원자 번호 범위 ──
_LANTHANIDE_Z = range(57, 72)   # La(57) ~ Lu(71)
_ACTINIDE_Z = range(89, 104)    # Ac(89) ~ Lr(103)
_HEAVY_ELEMENT_Z = set(_LANTHANIDE_Z) | set(_ACTINIDE_Z)


# ── Phase η-2: 수렴 에러 클래스 ──
class ConvergenceError(RuntimeError):
    """적응적 SCF 수렴 전략이 모두 실패했을 때 발생."""
    pass


# ── Phase η-2: 적응적 수렴 전략 ──
class ConvergenceStrategy:
    """적응적 SCF 수렴 에스컬레이션 엔진.

    5단계 전략을 순차적으로 시도하여 수렴시킨다:
    1. 기본 DIIS
    2. DIIS + level shift (0.5 Hartree)
    3. DIIS + level shift + damping
    4. SOSCF (Second-Order SCF, Newton)
    5. SOSCF + level shift
    """

    LEVELS = [
        {
            "name": "diis_default",
            "max_cycle": 100,
            "diis": True,
            "level_shift": 0.0,
            "soscf": False,
            "damp": 0.0,
        },
        {
            "name": "diis_levelshift",
            "max_cycle": 200,
            "diis": True,
            "level_shift": 0.5,
            "soscf": False,
            "damp": 0.0,
        },
        {
            "name": "diis_damp",
            "max_cycle": 200,
            "diis": True,
            "level_shift": 0.3,
            "soscf": False,
            "damp": 0.5,
        },
        {
            "name": "soscf",
            "max_cycle": 200,
            "diis": False,
            "level_shift": 0.0,
            "soscf": True,
            "damp": 0.0,
        },
        {
            "name": "soscf_shift",
            "max_cycle": 300,
            "diis": False,
            "level_shift": 0.5,
            "soscf": True,
            "damp": 0.0,
        },
    ]

    @staticmethod
    def apply(mf, level_idx: int = 0):
        """주어진 레벨의 수렴 전략을 mf 객체에 적용하고 반환."""
        if level_idx < 0 or level_idx >= len(ConvergenceStrategy.LEVELS):
            raise ValueError(f"Invalid strategy level: {level_idx}")
        cfg = ConvergenceStrategy.LEVELS[level_idx]
        mf.max_cycle = cfg["max_cycle"]
        mf.level_shift = cfg["level_shift"]
        mf.damp = cfg["damp"]
        if cfg["soscf"]:
            mf = mf.newton()
        return mf

    @staticmethod
    def level_name(level_idx: int) -> str:
        """전략 레벨의 이름 반환."""
        return ConvergenceStrategy.LEVELS[level_idx]["name"]


def _parse_atom_spec(atom_spec: str) -> str:
    """XYZ 파일 포맷 또는 PySCF 포맷을 모두 받아 PySCF atom 문자열로 변환."""
    lines = atom_spec.strip().splitlines()
    try:
        n_atoms = int(lines[0].strip())
    except ValueError:
        return atom_spec

    atom_lines: list[str] = []
    for line in lines[2:2 + n_atoms]:
        parts = line.split()
        if len(parts) >= 4:
            atom_lines.append(f"{parts[0]}  {parts[1]}  {parts[2]}  {parts[3]}")
    return "; ".join(atom_lines)


def _has_heavy_elements(mol) -> bool:
    """분자에 란타나이드/악티나이드 원소가 포함되어 있는지 확인."""
    if not _HAS_PYSCF:
        return False
    for ia in range(mol.natm):
        z = mol.atom_charge(ia)
        if int(z) in _HEAVY_ELEMENT_Z:
            return True
    return False


def _get_element_atomic_numbers(mol) -> list[int]:
    """분자의 모든 원자 번호 리스트 반환."""
    return [int(mol.atom_charge(ia)) for ia in range(mol.natm)]


class PySCFBackend(OrbitalBackend):
    """PySCF 기반 IAO/IBO 궤도 계산 및 체적 렌더링 백엔드.

    References:
        Knizia, J. Chem. Theory Comput. 2013, 9, 4834-4843.
        Knizia & Klein, Angew. Chem. Int. Ed. 2015, 54, 5518-5522.
    """

    @classmethod
    def name(cls) -> str:
        return "pyscf"

    @classmethod
    def is_available(cls) -> bool:
        return _HAS_PYSCF

    def compute_scf(self, atom_spec: str, basis: str = "cc-pvdz",
                    method: str = "RHF") -> SCFResult:
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        method_upper = method.upper()
        if method_upper not in _SUPPORTED_METHODS:
            raise ValueError(f"지원하지 않는 메서드 유형: {method}")

        atom_spec = _parse_atom_spec(atom_spec)
        mol = gto.M(atom=atom_spec, basis=basis, verbose=0)
        logger.info("SCF 시작: %d atoms, basis=%s, method=%s",
                     mol.natm, basis, method)

        if method_upper in ("RHF", "HF"):
            mf = scf.RHF(mol).run()
        elif method_upper == "UHF":
            mf = scf.UHF(mol).run()
        elif method_upper in ("RKS", "UKS", "B3LYP", "PBE0"):
            mf = scf.RKS(mol)
            if method_upper not in ("RKS", "UKS"):
                mf.xc = method
            mf = mf.run()
        else:
            raise ValueError(f"지원하지 않는 메서드 유형: {method}")

        if not mf.converged:
            raise RuntimeError(
                f"SCF calculation did not converge for '{atom_spec}' "
                f"with {method}/{basis}"
            )

        mo_coeff = mf.mo_coeff
        mo_occ = mf.mo_occ
        mo_energy = mf.mo_energy

        return SCFResult(
            converged=True,
            energy_hartree=float(mf.e_tot),
            mo_coeff=mo_coeff,
            mo_occ=mo_occ,
            mo_energy=mo_energy,
            basis=basis,
            method=method,
        ), mol

    # ── Phase η-1: X2C 스칼라 상대론 SCF ──

    def compute_scf_relativistic(
        self,
        atoms: list[str],
        coords: list[list[float]],
        basis: str | dict,
        ecp: str | dict | None = None,
        spin: int = 0,
        charge: int = 0,
        relativistic: str = "x2c",
    ):
        """X2C 스칼라 상대론 보정이 적용된 SCF 수행.

        Args:
            atoms: 원소 기호 리스트 (예: ['Ce', 'O']).
            coords: 좌표 리스트 (Angstrom). 각 항목은 [x, y, z].
            basis: 기저 함수 (str 또는 원소별 dict).
            ecp: ECP 설정 (str, dict, 또는 None).
            spin: 2S (홀전자 수).
            charge: 분자 전하.
            relativistic: 'x2c' 또는 'x2c1e' 또는 'none'.

        Returns:
            (mf, mol) 튜플.
        """
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        mol = gto.Mole()
        mol.atom = list(zip(atoms, coords))
        mol.basis = basis
        if ecp:
            mol.ecp = ecp
        mol.charge = charge
        mol.spin = spin
        mol.verbose = 0
        mol.build()

        if spin == 0:
            mf = scf.RHF(mol)
        else:
            mf = scf.UHF(mol)

        # 스칼라 상대론 적용
        if relativistic == "x2c":
            mf = mf.x2c()
        elif relativistic == "x2c1e":
            mf = mf.sfx2c1e()
        elif relativistic == "none":
            pass  # 비상대론
        else:
            raise ValueError(
                f"Unknown relativistic method: {relativistic}. "
                f"Use 'x2c', 'x2c1e', or 'none'."
            )

        logger.info(
            "Relativistic SCF 시작: %d atoms, spin=%d, relativistic=%s",
            mol.natm, spin, relativistic,
        )
        mf.kernel()

        if not mf.converged:
            logger.warning(
                "Relativistic SCF not converged, trying adaptive strategy..."
            )
            # 적응적 수렴 시도
            mf, mol = self._adaptive_relativistic_scf(
                mol, spin, relativistic
            )

        return mf, mol

    def _adaptive_relativistic_scf(self, mol, spin, relativistic):
        """상대론 SCF가 수렴하지 않을 때 적응적 전략 적용."""
        for level in range(1, len(ConvergenceStrategy.LEVELS)):
            if spin == 0:
                mf = scf.RHF(mol)
            else:
                mf = scf.UHF(mol)

            if relativistic == "x2c":
                mf = mf.x2c()
            elif relativistic == "x2c1e":
                mf = mf.sfx2c1e()

            mf = ConvergenceStrategy.apply(mf, level)
            try:
                mf.kernel()
                if mf.converged:
                    logger.info(
                        "Relativistic SCF converged at strategy level %d: %s",
                        level, ConvergenceStrategy.level_name(level),
                    )
                    return mf, mol
            except Exception as e:
                logger.warning("Relativistic strategy level %d failed: %s",
                               level, e)
                continue

        raise ConvergenceError(
            "Relativistic SCF failed to converge after all strategies."
        )

    def _resolve_minao(self, mol, minao="minao"):
        """ECP 감지 시 minao 자동 폴백.

        Phase η: 란타나이드/악티나이드 원소 감지 시 추가 경고 및 폴백.
        """
        warnings = []
        effective = minao
        ecp_detected = False

        if hasattr(mol, "has_ecp"):
            ecp_result = mol.has_ecp()
            ecp_detected = (
                bool(ecp_result)
                if not isinstance(ecp_result, dict)
                else len(ecp_result) > 0
            )
        if not ecp_detected and hasattr(mol, "_ecp") and mol._ecp:
            ecp_detected = True

        # ── Phase η: 란타나이드/악티나이드 특별 처리 ──
        has_heavy = _has_heavy_elements(mol)

        if ecp_detected and minao == "minao":
            effective = "sto-3g"
            warnings.append(
                "ECP detected. Switched IAO reference basis from 'minao' "
                "to 'sto-3g'. For best results with transition metals, "
                "use all-electron basis (e.g., def2-SVP) without ECP."
            )

        if has_heavy and effective == "minao":
            # 란타나이드가 있는데 ECP가 없는 경우 (all-electron X2C 등)
            # MINAO가 f-오비탈을 충분히 커버하지 못할 수 있음
            effective = "sto-3g"
            warnings.append(
                "Lanthanide/actinide element detected without ECP. "
                "Switched IAO reference basis from 'minao' to 'sto-3g'. "
                "MINAO may not have f-orbital coverage for heavy elements."
            )

        if has_heavy:
            atomic_numbers = _get_element_atomic_numbers(mol)
            heavy_z = [z for z in atomic_numbers if z in _HEAVY_ELEMENT_Z]
            logger.info(
                "Heavy elements detected (Z=%s). Using MINAO='%s' for IAO.",
                heavy_z, effective,
            )

        return effective, warnings

    @staticmethod
    def _unpack_uhf(mo_coeff, mo_occ):
        """UHF mo_coeff/mo_occ 언패킹."""
        import numpy as np
        if isinstance(mo_coeff, (tuple, list)):
            return mo_coeff[0], mo_coeff[1], mo_occ[0], mo_occ[1]
        elif isinstance(mo_coeff, np.ndarray) and mo_coeff.ndim == 3:
            return mo_coeff[0], mo_coeff[1], mo_occ[0], mo_occ[1]
        else:
            raise ValueError(
                f"Unexpected mo_coeff type: {type(mo_coeff)}, "
                f"ndim={getattr(mo_coeff, 'ndim', '?')}"
            )

    def compute_iao(self, scf_result: SCFResult, mol_obj: Any,
                    minao: str = "minao") -> IAOResult:
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        effective_minao, _ = self._resolve_minao(mol_obj, minao)

        orbocc = scf_result.mo_coeff[:, scf_result.mo_occ > 0]
        iao_coeff = lo.iao.iao(mol_obj, orbocc, minao=effective_minao)
        charges = self._compute_iao_charges(mol_obj, scf_result, iao_coeff)

        return IAOResult(
            coefficients=iao_coeff,
            charges=charges,
        )

    def compute_ibo(
        self,
        scf_result: SCFResult,
        iao_result: IAOResult,
        mol_obj: Any,
        localization_method: str = "PM",
    ) -> IBOResult:
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        orbocc = scf_result.mo_coeff[:, scf_result.mo_occ > 0]
        logger.info("IBO 계산 준비 중. 국소화 방법: %s", localization_method)

        if localization_method.upper() == "PM":
            ibo_obj = lo.ibo.ibo(mol_obj, orbocc,
                                  iaos=iao_result.coefficients)
            ibo_coeff = ibo_obj
        elif localization_method.upper() == "BOYS":
            boys_loc = lo.Boys(mol_obj, orbocc)
            boys_loc.verbose = 0
            boys_loc.kernel()
            ibo_coeff = boys_loc.mo_coeff
        else:
            raise ValueError(
                f"지원하지 않는 국소화 방식: {localization_method}"
            )

        n_ibo = ibo_coeff.shape[1]
        logger.info("IBO 계산 완료: %d IBOs 생성됨.", n_ibo)

        return IBOResult(
            coefficients=ibo_coeff,
            occupations=np.full(n_ibo, 2.0),
            n_ibo=n_ibo,
        )

    def _compute_iao_charges(self, mol: Any, scf_result: SCFResult,
                             iao_coeff: np.ndarray) -> np.ndarray:
        """IAO 기반 원자 부분 전하 — Löwdin 정규직교화 활용."""
        ovlp = mol.intor_symmetric("int1e_ovlp")
        orbocc = scf_result.mo_coeff[:, scf_result.mo_occ > 0]

        s_iao = iao_coeff.T @ ovlp @ iao_coeff
        eigvals, eigvecs = np.linalg.eigh(s_iao)
        s_iao_inv_half = (
            eigvecs @ np.diag(1.0 / np.sqrt(eigvals)) @ eigvecs.T
        )
        iao_orth = iao_coeff @ s_iao_inv_half

        proj = iao_orth.T @ ovlp @ orbocc
        dm_iao = 2.0 * proj @ proj.T

        ao_labels = mol.ao_labels(fmt=False)
        n_iao = iao_orth.shape[1]
        charges = np.zeros(mol.natm)

        for j in range(n_iao):
            atom_idx = ao_labels[j][0]
            charges[atom_idx] += dm_iao[j, j]

        for i in range(mol.natm):
            charges[i] = mol.atom_charge(i) - charges[i]

        return charges

    def generate_cube(
        self,
        mol_obj: Any,
        orbital_coeff: np.ndarray,
        orbital_index: int,
        grid_points: tuple[int, int, int] = (80, 80, 80),
    ) -> str:
        """Gaussian cube 포맷 문자열을 반환."""
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        import tempfile
        import os

        coeff_vector = orbital_coeff[:, orbital_index]

        tmp_path = None
        try:
            fd, tmp_path = tempfile.mkstemp(suffix=".cube")
            os.close(fd)

            cubegen.orbital(
                mol_obj,
                tmp_path,
                coeff_vector,
                nx=grid_points[0],
                ny=grid_points[1],
                nz=grid_points[2],
            )

            with open(tmp_path, "r") as f:
                cube_text = f.read()

            logger.info(
                "Cube 생성 완료: orbital_index=%d, grid=%s, size=%d bytes",
                orbital_index, grid_points, len(cube_text),
            )
            return cube_text

        finally:
            if tmp_path and os.path.exists(tmp_path):
                os.remove(tmp_path)

    def export_molden(
        self,
        mol_obj: Any,
        mo_coeff: np.ndarray,
        output_path: str,
    ) -> str:
        """IBO/IAO/canonical 궤도를 Molden 포맷으로 내보내기."""
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        from pyscf.tools import molden as molden_mod

        molden_mod.from_mo(mol_obj, output_path, mo_coeff)
        logger.info("Molden 파일 생성: %s (%d orbitals)",
                     output_path, mo_coeff.shape[1])
        return str(Path(output_path).resolve())

    # ── UHF/ROHF 지원 (Phase ζ-2) ──

    def compute_scf_flexible(self, atom_spec: str, basis: str = "sto-3g",
                             charge: int = 0, spin: int = 0,
                             adaptive: bool = False):
        """spin>0 시 UHF 자동 선택. (mf, mol) 튜플 반환.

        Phase η: adaptive=True 시 적응적 수렴 에스컬레이션 사용.
        """
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")
        atom_spec = _parse_atom_spec(atom_spec)
        mol = gto.M(atom=atom_spec, basis=basis, charge=charge,
                     spin=spin, verbose=0)

        if adaptive:
            return self.compute_scf_adaptive(mol, spin=spin)

        if spin > 0:
            mf = scf.UHF(mol)
        else:
            mf = scf.RHF(mol)
        mf.kernel()
        if not mf.converged:
            raise RuntimeError(f"SCF not converged for spin={spin}")
        return mf, mol

    # ── Phase η-2: 적응적 SCF 수렴 ──

    def compute_scf_adaptive(self, mol, spin: int = 0,
                             max_escalation: int = 4):
        """자동 에스컬레이션으로 SCF 수렴을 시도.

        Args:
            mol: PySCF Mole 객체 (이미 build된 상태).
            spin: 2S (홀전자 수). 0이면 RHF, >0이면 UHF.
            max_escalation: 최대 에스컬레이션 레벨 (0-4).

        Returns:
            (mf, mol) 튜플.

        Raises:
            ConvergenceError: 모든 전략 실패 시.
        """
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        max_level = min(max_escalation,
                        len(ConvergenceStrategy.LEVELS) - 1)

        for level in range(max_level + 1):
            if spin == 0:
                mf = scf.RHF(mol)
            else:
                mf = scf.UHF(mol)

            mf = ConvergenceStrategy.apply(mf, level)

            try:
                mf.kernel()
                if mf.converged:
                    strategy_name = ConvergenceStrategy.level_name(level)
                    logger.info(
                        "SCF converged at strategy level %d: %s "
                        "(E=%.8f Hartree)",
                        level, strategy_name, mf.e_tot,
                    )

                    # 안정성 분석 (UHF에서만)
                    if spin > 0:
                        try:
                            mo_new = mf.stability()[0]
                            if not np.allclose(mo_new, mf.mo_coeff):
                                logger.warning(
                                    "Wavefunction instability detected, "
                                    "re-running with new initial guess..."
                                )
                                dm_new = mf.make_rdm1(mo_coeff=mo_new)
                                mf.kernel(dm0=dm_new)
                                if mf.converged:
                                    logger.info(
                                        "SCF re-converged after stability "
                                        "analysis (E=%.8f Hartree)",
                                        mf.e_tot,
                                    )
                        except Exception as e:
                            logger.debug(
                                "Stability analysis skipped: %s", e
                            )

                    return mf, mol
                else:
                    logger.info(
                        "Strategy level %d (%s) did not converge.",
                        level, ConvergenceStrategy.level_name(level),
                    )
            except Exception as e:
                logger.warning(
                    "Strategy level %d (%s) failed: %s",
                    level, ConvergenceStrategy.level_name(level), e,
                )
                continue

        raise ConvergenceError(
            f"SCF failed to converge after {max_level + 1} strategies. "
            f"Last strategy: "
            f"{ConvergenceStrategy.level_name(max_level)}"
        )

    def compute_iao_uhf(self, mf, mol, minao: str = "minao"):
        """UHF IAO: alpha/beta 스핀 채널 분리."""
        effective, warnings = self._resolve_minao(mol, minao)
        mo_a, mo_b, occ_a, occ_b = self._unpack_uhf(
            mf.mo_coeff, mf.mo_occ
        )
        mo_occ_a = mo_a[:, occ_a > 0]
        mo_occ_b = mo_b[:, occ_b > 0]
        iao_a = lo.iao.iao(mol, mo_occ_a, minao=effective)
        iao_b = lo.iao.iao(mol, mo_occ_b, minao=effective)
        return {
            "alpha": {"iao_coeff": iao_a, "n_iao": iao_a.shape[1]},
            "beta":  {"iao_coeff": iao_b, "n_iao": iao_b.shape[1]},
            "is_uhf": True,
            "minao_used": effective,
            "warnings": warnings,
        }

    def compute_ibo_uhf(self, mf, iao_result, mol):
        """UHF IBO: alpha/beta 각각 로컬라이즈."""
        mo_a, mo_b, occ_a, occ_b = self._unpack_uhf(
            mf.mo_coeff, mf.mo_occ
        )
        mo_occ_a = mo_a[:, occ_a > 0]
        mo_occ_b = mo_b[:, occ_b > 0]
        ibo_a = lo.ibo.ibo(
            mol, mo_occ_a, iaos=iao_result["alpha"]["iao_coeff"]
        )
        ibo_b = lo.ibo.ibo(
            mol, mo_occ_b, iaos=iao_result["beta"]["iao_coeff"]
        )
        return {
            "alpha": {"ibo_coeff": ibo_a, "n_ibo": ibo_a.shape[1]},
            "beta":  {"ibo_coeff": ibo_b, "n_ibo": ibo_b.shape[1]},
            "is_uhf": True,
            "total_ibo": ibo_a.shape[1] + ibo_b.shape[1],
        }

    def compute_uhf_charges(self, mf, mol):
        """UHF Mulliken 전하."""
        dm = mf.make_rdm1()
        s = mol.intor("int1e_ovlp")
        dm_total = dm[0] + dm[1] if isinstance(dm, (list, tuple)) else dm
        pop = np.einsum("ij,ji->i", dm_total, s)
        charges = []
        offset = 0
        for ia in range(mol.natm):
            nao_atom = sum(
                (2 * mol.bas_angular(ib) + 1) * mol.bas_nctr(ib)
                for ib in range(mol.nbas)
                if mol.bas_atom(ib) == ia
            )
            q = mol.atom_charge(ia) - sum(pop[offset:offset + nao_atom])
            charges.append(float(q))
            offset += nao_atom
        return charges

registry.register(PySCFBackend)
```

```python
Playwright+Chromium (~200MB) 대신 PyVista+vtk-osmesa (~50MB)를 사용하여

from pathlib import Path

    import pyvista as pv

def cube_to_pyvista_grid(
    cube_data: np.ndarray,
    origin: tuple[float, float, float],
    axes: list[np.ndarray],
    npts: tuple[int, int, int],
) -> "pv.ImageData":
    """PySCF 큐브 데이터를 PyVista ImageData로 변환.

    Args:
        cube_data: 3D numpy array (nx, ny, nz) 형태의 오비탈 값.

        float(np.linalg.norm(ax) / max(n - 1, 1))
        for ax, n in zip(axes, npts)
    )

    grid = pv.ImageData(
        dimensions=npts,
        spacing=spacing,
        origin=origin,
    )
    grid["orbital"] = cube_data.flatten(order="F")
    return grid
```

```python
def verify_basis_independence(
    molecule_name: str,
    results_by_basis: dict[str, dict],
) -> dict:
    """여러 기저의 IBO 결과를 비교하여 기저 독립성 검증.

    Knizia 2013 원논문에 따르면, IAO 투영 덕분에 IBO 수와
    화학적 해석이 기저 함수에 독립적이어야 한다.

    Args:
        molecule_name: 분자 이름 (로깅용).
        results_by_basis: {basis_name: {"n_ibo": int, "charges": np.ndarray}}

    검증 기준:
    - IBO 수: 모든 기저에서 동일
    - 전하 합: 모든 기저에서 |Σq| < 1e-4
    - 원자별 전하 편차: 기저 간 max |Δq| < 0.15 e

    Returns:
        dict: {
            "ibo_count_invariant": bool,
            "charge_conservation": bool,
            "charge_deviation_ok": bool,
            "max_charge_deviation": float,
            "ibo_counts": {basis: n_ibo},
            "all_passed": bool,
        }
    """
    if len(results_by_basis) < 2:
        return {
            "ibo_count_invariant": True,
            "charge_conservation": True,
            "charge_deviation_ok": True,
            "max_charge_deviation": 0.0,
            "ibo_counts": {
                b: r["n_ibo"] for b, r in results_by_basis.items()
            },
            "all_passed": True,
        }

    # IBO 수 비교
    ibo_counts = {b: r["n_ibo"] for b, r in results_by_basis.items()}
    unique_counts = set(ibo_counts.values())
    ibo_invariant = len(unique_counts) == 1

    # 전하 합 확인
    charge_conservation = True
    for b, r in results_by_basis.items():
        if "charges" in r and r["charges"] is not None:
            charge_sum = float(np.sum(r["charges"]))
            if abs(charge_sum) >= 1e-4:
                charge_conservation = False
                logger.warning(
                    "%s/%s: charge sum = %.6f (expected ~0)",
                    molecule_name, b, charge_sum,
                )

    # 원자별 전하 편차 (기저 간)
    charge_arrays = []
    for b, r in results_by_basis.items():
        if "charges" in r and r["charges"] is not None:
            charge_arrays.append(r["charges"])

    max_deviation = 0.0
    if len(charge_arrays) >= 2:
        # 모든 쌍 비교
        for i in range(len(charge_arrays)):
            for j in range(i + 1, len(charge_arrays)):
                if len(charge_arrays[i]) == len(charge_arrays[j]):
                    dev = float(np.max(np.abs(
                        charge_arrays[i] - charge_arrays[j]
                    )))
                    max_deviation = max(max_deviation, dev)

    charge_deviation_ok = max_deviation < 0.15

    all_passed = ibo_invariant and charge_conservation and charge_deviation_ok

    result = {
        "ibo_count_invariant": ibo_invariant,
        "charge_conservation": charge_conservation,
        "charge_deviation_ok": charge_deviation_ok,
        "max_charge_deviation": max_deviation,
        "ibo_counts": ibo_counts,
        "all_passed": all_passed,
    }

    if not all_passed:
        logger.warning(
            "%s basis independence check FAILED: %s", molecule_name, result,
        )
    else:
        logger.info(
            "%s basis independence check PASSED (max Δq=%.4f)",
            molecule_name, max_deviation,
        )

    return result
```

```python
            except Exception as e:
                elapsed = time.time() - t0
                total_skipped += 1
                print(
                    f"{mol.name:<20} {basis:<16} "
                    f"{'SKIP':<14} {'---':<17} "
                    f"{'---':<12} {elapsed:.1f}s  SKIP ({str(e)[:40]})"
                )
                all_results.append({
                    "name": mol.name,
                    "basis": basis,
                    "status": "skip",
                    "error": str(e)[:100],
                    "elapsed_s": round(elapsed, 2),
                })

        # 기저 독립성 검증
        if len(mol_results) >= 2:
            ind_result = verify_basis_independence(mol.name, mol_results)
            independence_results.append({
                "molecule": mol.name,
                **{k: v for k, v in ind_result.items()
                   if k != "ibo_counts"},
                "ibo_counts": {
                    b: int(c) for b, c in ind_result["ibo_counts"].items()
                },
            })

    print(sep)
    total = total_passed + total_failed + total_skipped
    print(
        f"Total: {total} runs, "
        f"{total_passed} passed, "
        f"{total_failed} failed, "
        f"{total_skipped} skipped"
    )

    # 기저 독립성 요약
    print(f"\n{'='*50}")
    print("Basis Independence Summary")
    print(f"{'='*50}")
    for ir in independence_results:
        status = "✅ PASS" if ir["all_passed"] else "❌ FAIL"
        print(
            f"  {ir['molecule']:<20} {status}  "
            f"IBO counts: {ir['ibo_counts']}  "
            f"max Δq: {ir['max_charge_deviation']:.4f}"
        )

    # 결과 저장
    RESULTS_DIR.mkdir(parents=True, exist_ok=True)
    summary = {
        "results": all_results,
        "independence": independence_results,
        "summary": {
            "total": total,
            "passed": total_passed,
            "failed": total_failed,
            "skipped": total_skipped,
        },
    }
    summary_path = RESULTS_DIR / "extended_summary.json"
    summary_path.write_text(
        json.dumps(summary, indent=2, ensure_ascii=False, default=str)
    )
    print(f"\nResults saved to {summary_path}")

    return 0 if total_failed == 0 else 1


if __name__ == "__main__":
    sys.exit(main())
```

```python
"""Phase η-4: 기저 함수 확장 벤치마크 및 독립성 검증 테스트.

IBO 수가 기저 함수에 독립적인지 (Knizia 2013) 정량적으로 검증한다.

import time

import pytest
import numpy as np

pyscf = pytest.importorskip("pyscf")

from qcviz_mcp.backends.pyscf_backend import PySCFBackend
from qcviz_mcp.validation import verify_basis_independence

backend = PySCFBackend()

# ── 테스트용 분자 정의 ──
WATER = "O 0 0 0; H 0 0.757 0.587; H 0 -0.757 0.587"
METHANE = (
    "C 0 0 0; "
    "H  0.629  0.629  0.629; "
    "H -0.629 -0.629  0.629; "
    "H -0.629  0.629 -0.629; "
    "H  0.629 -0.629 -0.629"

ETHYLENE = (
    "C 0 0  0.667; C 0 0 -0.667; "
    "H 0  0.923  1.237; H 0 -0.923  1.237; "
    "H 0  0.923 -1.237; H 0 -0.923 -1.237"
)


def _compute_ibo_for_basis(atom_spec: str, basis: str) -> dict:
    """주어진 기저에서 IBO 계산 후 결과 dict 반환."""
    scf_res, mol_obj = backend.compute_scf(atom_spec, basis=basis, method="HF")
    iao_res = backend.compute_iao(scf_res, mol_obj)
    ibo_res = backend.compute_ibo(scf_res, iao_res, mol_obj)
    return {
        "n_ibo": ibo_res.n_ibo,
        "charges": iao_res.charges,
        "energy": scf_res.energy_hartree,
    }


class TestBasisIndependence:
    """IBO 기저 독립성 검증."""

    def test_water_ibo_count_invariant_fast(self):
        """물 IBO 수 = 5 (sto-3g, cc-pvdz)."""
        for basis in ("sto-3g", "cc-pvdz"):
            result = _compute_ibo_for_basis(WATER, basis)
            assert result["n_ibo"] == 5, (
                f"Water/{basis}: expected 5, got {result['n_ibo']}"
            )

    @pytest.mark.slow
    def test_water_ibo_count_invariant(self):

        for basis in ("sto-3g", "cc-pvdz", "cc-pvtz", "aug-cc-pvdz"):
            try:
                result = _compute_ibo_for_basis(WATER, basis)
                assert result["n_ibo"] == 5, (
                    f"Water/{basis}: expected 5, got {result['n_ibo']}"
                )
            except Exception as e:
                if "basis" in str(e).lower():
                    pytest.skip(f"Basis {basis} not available: {e}")
                raise

    @pytest.mark.slow
    def test_methane_ibo_count_invariant(self):
        """메탄 IBO 수 = 5 (4가지 기저)."""
        for basis in ("sto-3g", "cc-pvdz", "cc-pvtz", "aug-cc-pvdz"):
            try:
                result = _compute_ibo_for_basis(METHANE, basis)
                assert result["n_ibo"] == 5, (
                    f"Methane/{basis}: expected 5, got {result['n_ibo']}"
                )
            except Exception as e:
                if "basis" in str(e).lower():
                    pytest.skip(f"Basis {basis} not available: {e}")
                raise

    @pytest.mark.slow
    def test_ethylene_ibo_count_invariant(self):
        """에틸렌 IBO 수 = 8 (4가지 기저)."""
        for basis in ("sto-3g", "cc-pvdz", "cc-pvtz", "aug-cc-pvdz"):
            try:

                assert result["n_ibo"] == 8, (

                )
            except Exception as e:
                if "basis" in str(e).lower():
                    pytest.skip(f"Basis {basis} not available: {e}")
                raise

    @pytest.mark.slow
    def test_charge_conservation_all_bases(self):
        """전하 합 ≈ 0 (물, 메탄, 에틸렌 × 4가지 기저)."""
        molecules = {
            "water": WATER,
            "methane": METHANE,
            "ethylene": ETHYLENE,
        }
        for mol_name, atom_spec in molecules.items():
            for basis in ("sto-3g", "cc-pvdz", "cc-pvtz", "aug-cc-pvdz"):
                try:
                    result = _compute_ibo_for_basis(atom_spec, basis)
                    charge_sum = float(np.sum(result["charges"]))
                    assert abs(charge_sum) < 1e-4, (
                        f"{mol_name}/{basis}: charge sum = {charge_sum:.6f}"
                    )
                except Exception as e:
                    if "basis" in str(e).lower():
                        continue  # 기저 없으면 skip
                    raise

    @pytest.mark.slow
    def test_charge_deviation_small(self):
        """원자별 전하 편차 < 0.15 e (기저 간)."""
        results_by_basis = {}
        for basis in ("sto-3g", "cc-pvdz", "cc-pvtz"):
            try:
                results_by_basis[basis] = _compute_ibo_for_basis(WATER, basis)
            except Exception:
                continue

        if len(results_by_basis) < 2:
            pytest.skip("Not enough bases available for comparison")

        ind = verify_basis_independence("water", results_by_basis)
        assert ind["charge_deviation_ok"], (

            f"(threshold: 0.15)"

            t_ccpvtz = time.time() - t0
        except Exception as e:

                pytest.skip(f"cc-pvtz not available: {e}")
            raise

        ratio = t_ccpvtz / max(t_sto3g, 0.001)
        assert ratio < 100, (
            f"cc-pVTZ is {ratio:.1f}x slower than STO-3G (limit: 100x)"
```

## Phase η 구현 계획 평가

전체적으로 잘 구조화된 계획이지만, 실제 실행 전에 짚어야 할 문제점들이 있습니다. 크게 세 가지 범주로 나누어 봅니다.

---

### 좋은 점

**설계 원칙이 일관됩니다.** 모든 새 의존성에 `try/except ImportError` 가드를 붙이고, 느린 테스트는 `@pytest.mark.slow`로 분리하고, 기존 108 테스트를 건드리지 않겠다는 방침이 전체에 걸쳐 지켜지고 있습니다.

**η-2 적응적 수렴 엔진은 실용적입니다.** 5단계 에스컬레이션(DIIS → level shift → damp → SOSCF → SOSCF+shift)은 PySCF 실무에서 실제로 쓰는 패턴이고, `ConvergenceError`로 전략 소진을 명시적으로 처리하는 것도 깔끔합니다. UHF stability analysis까지 포함한 것은 실질적으로 가치가 있습니다.

**η-3 PyVista 대안은 올바른 방향입니다.** Playwright+Chromium 200MB → PyVista+vtk-osmesa ~50MB로 경량화하고, `get_best_renderer()`로 PyVista > Playwright > HTML-only 자동 선택하는 구조가 합리적입니다.

**η-4 기저 독립성 검증 프레임워크도 좋습니다.** Knizia 2013 원논문의 핵심 주장(IBO 수가 기저에 무관)을 정량적으로 검증하는 `verify_basis_independence()`는 JOSS 논문의 Evidence 섹션에 직접 쓸 수 있는 결과를 생성합니다.

### 문제점 — 실행 전에 반드시 수정해야 하는 것들

**1. pyscf_backend.py 전체 교체는 위험합니다.**

"기존 파일 전체를 아래로 교체합니다"라고 되어 있는데, 이 파일은 프로젝트의 핵심 파일입니다. Phase ζ에서 추가한 코드(예: `compute_scf_flexible`의 기존 로직, 보안 검증 함수 `_validate_file_path`, `_validate_atom_spec`)가 이 교체본에 빠져 있을 가능성이 높습니다. 실제로 교체본에는 `_validate_file_path`, `_validate_atom_spec`이 보이지 않습니다. 이것은 Phase δ에서 추가한 보안 패치를 날리는 결과를 초래합니다. **전체 교체가 아니라 diff/patch 방식으로 추가 부분만 삽입해야 합니다.**

**2. 란타나이드 테스트(η-1)의 실행 가능성이 불확실합니다.**

CeO, GdF₃ 테스트가 `@pytest.mark.slow` + `pytest.skip` 폴백으로 방어되어 있긴 하지만, PySCF의 내장 LANL2DZ가 Ce(Z=58), Gd(Z=64)를 실제로 지원하는지 확인이 안 된 상태입니다. PySCF의 LANL2DZ ECP는 주로 3d/4d 전이금속까지만 내장되어 있고, 란타나이드는 별도 BSE(Basis Set Exchange) 다운로드가 필요할 수 있습니다. 테스트 7개 중 4~5개가 전부 skip되면 이 Phase의 실질적 검증 가치가 크게 떨어집니다. 이 부분은 솔직히 **실행해보기 전까지는 모릅니다.**

**3. `compute_scf_relativistic`의 인터페이스가 기존 `compute_scf`와 불일치합니다.**

기존 `compute_scf`는 `atom_spec: str` (PySCF 문자열)을 받는데, 새 `compute_scf_relativistic`은 `atoms: list[str]` + `coords: list[list[float]]`을 받습니다. 같은 백엔드 안에서 인터페이스 규약이 달라지면 혼란을 줍니다. 하나로 통일하거나, 내부에서 변환 레이어를 두는 게 낫습니다.

**4. `ConvergenceStrategy.LEVELS`가 클래스 변수(mutable list)입니다.**

`test_convergence_error_raised`에서 `ConvergenceStrategy.LEVELS[0]["max_cycle"] = 1`로 직접 수정한 뒤 `finally`로 복원하는 패턴이 있는데, 이것은 병렬 테스트 실행 시 race condition을 유발합니다. 테스트에서 전역 상태를 변경하는 것은 피해야 합니다. `monkeypatch`를 쓰거나 `max_escalation=0` + 불가능한 분자 조합으로 실패를 유도하는 게 더 안전합니다.

**5. PyVista 렌더러의 큐브 데이터 입력 형식이 PySCF `generate_cube`와 연결되지 않습니다.**

`render_orbital_png`이 `(cube_data: np.ndarray, origin, axes, npts)`를 받는데, 기존 `generate_cube` 메서드는 큐브를 **문자열(Gaussian cube format)**로 반환합니다. 큐브 문자열 → numpy 3D array 파싱 함수가 없으면 실제 파이프라인에서 연결이 안 됩니다. 테스트에서는 합성 데이터(가우시안 함수)로 우회했기 때문에 통과하겠지만, 실제 IBO → PNG 경로는 동작하지 않습니다. **`parse_cube_string()` 같은 변환 함수가 필요합니다.**

**6. `benchmark/molecules.py` 수정이 기존 구조와 충돌할 수 있습니다.**

`LanthanideBenchmarkMolecule`이라는 새 dataclass를 추가하고 `EXTENDED_BASIS_SETS` 상수를 추가하는데, 기존 `MOLECULES` 리스트의 구조(아마 다른 dataclass)와 `run_extended_benchmark.py`에서 `MOLECULES`를 참조하는 방식이 호환되는지 확인이 필요합니다. `run_extended_benchmark.py`에서 `mol.atom_spec`을 호출하는데, `LanthanideBenchmarkMolecule`에는 `atom_spec` 필드가 없습니다.

**A. 파일 상단 (import 직후, `_SUPPORTED_METHODS` 바로 아래)에 추가:**

```python
# ── Phase η: 무거운 원소 감지 + 수렴 에스컬레이션 ──

_HEAVY_TM_Z = set(range(39, 49)) | set(range(72, 81))  # 4d(Y-Cd) + 5d(Hf-Hg)


class ConvergenceError(RuntimeError):
    """적응적 SCF 수렴 전략이 모두 실패했을 때 발생."""
    pass


class ConvergenceStrategy:
    """적응적 SCF 수렴 에스컬레이션 엔진.

    5단계 전략을 순차적으로 시도:
    1. 기본 DIIS
    2. DIIS + level shift (0.5 Hartree)
    3. DIIS + level shift + damping
    4. SOSCF (Second-Order SCF, Newton)
    5. SOSCF + level shift
    """

    LEVELS = (
        {"name": "diis_default",   "max_cycle": 100, "level_shift": 0.0, "soscf": False, "damp": 0.0},
        {"name": "diis_levelshift","max_cycle": 200, "level_shift": 0.5, "soscf": False, "damp": 0.0},
        {"name": "diis_damp",      "max_cycle": 200, "level_shift": 0.3, "soscf": False, "damp": 0.5},
        {"name": "soscf",          "max_cycle": 200, "level_shift": 0.0, "soscf": True,  "damp": 0.0},
        {"name": "soscf_shift",    "max_cycle": 300, "level_shift": 0.5, "soscf": True,  "damp": 0.0},
    )

    @staticmethod
    def apply(mf, level_idx: int = 0):
        """주어진 레벨의 수렴 전략을 mf 객체에 적용하고 반환."""
        if level_idx < 0 or level_idx >= len(ConvergenceStrategy.LEVELS):
            raise ValueError(f"Invalid strategy level: {level_idx}")
        cfg = ConvergenceStrategy.LEVELS[level_idx]
        mf.max_cycle = cfg["max_cycle"]
        mf.level_shift = cfg["level_shift"]
        mf.damp = cfg["damp"]
        if cfg["soscf"]:
            mf = mf.newton()
        return mf

    @staticmethod
    def level_name(level_idx: int) -> str:
        return ConvergenceStrategy.LEVELS[level_idx]["name"]


def _has_heavy_tm(mol) -> bool:
    """분자에 4d/5d 전이금속이 포함되어 있는지 확인."""
    if not _HAS_PYSCF:
        return False
    for ia in range(mol.natm):
        if int(mol.atom_charge(ia)) in _HEAVY_TM_Z:
            return True
    return False


def parse_cube_string(cube_text: str) -> dict:
    """Gaussian cube 포맷 문자열을 파싱하여 numpy 3D array + 메타데이터 반환.

    기존 generate_cube() → 문자열 → parse_cube_string() → PyVista 렌더러
    파이프라인 연결용.

    Args:
        cube_text: Gaussian cube format 문자열.

    Returns:
        dict: {
            "data": np.ndarray (nx, ny, nz),
            "origin": (x, y, z) in Bohr,
            "axes": [dx_vec, dy_vec, dz_vec] — 각 축 벡터 (3,),
            "npts": (nx, ny, nz),
            "atoms": [(Z, x, y, z), ...],
        }
    """
    lines = cube_text.strip().splitlines()

    # 첫 두 줄은 코멘트
    # 줄2: 원자 수, origin_x, origin_y, origin_z
    parts = lines[2].split()
    natm = abs(int(parts[0]))
    origin = (float(parts[1]), float(parts[2]), float(parts[3]))

    # 줄3~5: 각 축의 그리드 수 + 축 벡터
    axes = []
    npts_list = []
    for i in range(3):
        p = lines[3 + i].split()
        n = int(p[0])
        npts_list.append(n)
        vec = np.array([float(p[1]), float(p[2]), float(p[3])]) * n
        axes.append(vec)
    npts = tuple(npts_list)

    # 원자 데이터 (줄6 ~ 6+natm-1)
    atoms = []
    for i in range(natm):
        p = lines[6 + i].split()
        atoms.append((int(float(p[0])), float(p[2]), float(p[3]), float(p[4])))

    # 볼류메트릭 데이터
    data_start = 6 + natm
    values = []
    for line in lines[data_start:]:
        values.extend(float(v) for v in line.split())

    data = np.array(values).reshape(npts)

    return {
        "data": data,
        "origin": origin,
        "axes": axes,
        "npts": npts,
        "atoms": atoms,
    }
```

**B. `PySCFBackend` 클래스 내부 (맨 마지막, `compute_uhf_charges` 뒤)에 추가:**

```python
    # ── Phase η-1: X2C 스칼라 상대론 SCF ──

    def compute_scf_relativistic(
        self,
        atom_spec: str,
        basis: str | dict = "def2-svp",
        ecp: str | dict | None = None,
        spin: int = 0,
        charge: int = 0,
        relativistic: str = "sfx2c1e",
    ):
        """X2C 스칼라 상대론 보정이 적용된 SCF 수행.

        Args:
            atom_spec: 분자 좌표 (PySCF 포맷 또는 XYZ 포맷).
            basis: 기저 함수 (str 또는 원소별 dict).
            ecp: ECP 설정 (str, dict, 또는 None).
            spin: 2S (홀전자 수).
            charge: 분자 전하.
            relativistic: 'sfx2c1e' 또는 'x2c' 또는 'none'.

        Returns:
            (mf, mol) 튜플.
        """
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        atom_spec = _parse_atom_spec(atom_spec)
        mol = gto.Mole()
        mol.atom = atom_spec
        mol.basis = basis
        if ecp:
            mol.ecp = ecp
        mol.charge = charge
        mol.spin = spin
        mol.verbose = 0
        mol.build()

        if spin == 0:
            mf = scf.RHF(mol)
        else:
            mf = scf.UHF(mol)

        if relativistic == "sfx2c1e":
            mf = mf.sfx2c1e()
        elif relativistic == "x2c":
            mf = mf.x2c()
        elif relativistic == "none":
            pass
        else:
            raise ValueError(
                f"Unknown relativistic method: {relativistic}. "
                f"Use 'sfx2c1e', 'x2c', or 'none'."
            )

        logger.info(
            "Relativistic SCF 시작: %d atoms, spin=%d, relativistic=%s",
            mol.natm, spin, relativistic,
        )
        mf.kernel()

        if not mf.converged:
            logger.warning("Relativistic SCF not converged, trying adaptive...")
            mf, mol = self._run_adaptive(mol, spin, relativistic)

        return mf, mol

    def _run_adaptive(self, mol, spin: int, relativistic: str = "none"):
        """SCF 미수렴 시 적응적 전략 순차 시도."""
        for level in range(1, len(ConvergenceStrategy.LEVELS)):
            if spin == 0:
                mf = scf.RHF(mol)
            else:
                mf = scf.UHF(mol)

            if relativistic == "sfx2c1e":
                mf = mf.sfx2c1e()
            elif relativistic == "x2c":
                mf = mf.x2c()

            mf = ConvergenceStrategy.apply(mf, level)
            try:
                mf.kernel()
                if mf.converged:
                    logger.info(
                        "SCF converged at strategy level %d: %s (E=%.8f)",
                        level, ConvergenceStrategy.level_name(level), mf.e_tot,
                    )
                    return mf, mol
            except Exception as e:
                logger.warning("Strategy level %d failed: %s", level, e)
                continue

        raise ConvergenceError(
            "SCF failed to converge after all adaptive strategies."
        )

    # ── Phase η-2: 적응적 SCF (비상대론 경로 통합) ──

    def compute_scf_adaptive(self, mol, spin: int = 0, max_escalation: int = 4):
        """자동 에스컬레이션으로 SCF 수렴 시도.

        Args:
            mol: PySCF Mole 객체 (이미 build된 상태).
            spin: 2S. 0이면 RHF, >0이면 UHF.
            max_escalation: 최대 레벨 (0-4).

        Returns:
            (mf, mol) 튜플.

        Raises:
            ConvergenceError: 모든 전략 실패 시.
        """
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")

        max_level = min(max_escalation, len(ConvergenceStrategy.LEVELS) - 1)

        for level in range(max_level + 1):
            if spin == 0:
                mf = scf.RHF(mol)
            else:
                mf = scf.UHF(mol)

            mf = ConvergenceStrategy.apply(mf, level)

            try:
                mf.kernel()
                if mf.converged:
                    logger.info(
                        "SCF converged at strategy level %d: %s (E=%.8f)",
                        level, ConvergenceStrategy.level_name(level), mf.e_tot,
                    )
                    return mf, mol
                else:
                    logger.info(
                        "Strategy level %d (%s) did not converge.",
                        level, ConvergenceStrategy.level_name(level),
                    )
            except Exception as e:
                logger.warning("Strategy level %d failed: %s", level, e)
                continue

        raise ConvergenceError(
            f"SCF failed to converge after {max_level + 1} strategies."
        )
```

**C. `compute_scf_flexible` 메서드 수정 (기존 메서드를 교체):**

기존 `compute_scf_flexible` 메서드의 본문만 교체합니다. 시그니처에 `adaptive` 파라미터를 추가합니다.

```python
    def compute_scf_flexible(self, atom_spec: str, basis: str = "sto-3g",
                             charge: int = 0, spin: int = 0,
                             adaptive: bool = False):
        """spin>0 시 UHF 자동 선택. (mf, mol) 튜플 반환.

        Phase η: adaptive=True 시 적응적 수렴 에스컬레이션 사용.
        """
        if not _HAS_PYSCF:
            raise ImportError("PySCF가 설치되지 않았습니다.")
        atom_spec = _parse_atom_spec(atom_spec)
        mol = gto.M(atom=atom_spec, basis=basis, charge=charge,
                     spin=spin, verbose=0)

        if adaptive:
            return self.compute_scf_adaptive(mol, spin=spin)

        if spin > 0:
            mf = scf.UHF(mol)
        else:
            mf = scf.RHF(mol)
        mf.kernel()
        if not mf.converged:
            raise RuntimeError(f"SCF not converged for spin={spin}")
        return mf, mol
```

> **적용 방법:** `compute_scf_flexible` 메서드를 찾아서 본문만 위 코드로 교체합니다. 다른 메서드는 건드리지 않습니다.

```python
    def _resolve_minao(self, mol, minao='minao'):
        """ECP 감지 시 minao 자동 폴백.

        Phase η: 4d/5d 전이금속 감지 시 추가 경고.
        """
        warnings = []
        effective = minao
        ecp_detected = False
        if hasattr(mol, 'has_ecp'):
            ecp_result = mol.has_ecp()
            ecp_detected = bool(ecp_result) if not isinstance(ecp_result, dict) else len(ecp_result) > 0
        if not ecp_detected and hasattr(mol, '_ecp') and mol._ecp:
            ecp_detected = True

        # Phase η: 4d/5d 전이금속 감지
        has_heavy = _has_heavy_tm(mol)

        if ecp_detected and minao == 'minao':
            effective = 'sto-3g'
            warnings.append(
                "ECP detected. Switched IAO reference basis from 'minao' to 'sto-3g'. "
                "For best results with transition metals, use all-electron "
                "basis (e.g., def2-SVP) without ECP."
            )

        if has_heavy and not ecp_detected and minao == 'minao':
            warnings.append(
                "4d/5d transition metal detected (all-electron). "
                "MINAO may lack coverage for high-angular-momentum shells. "
                "Consider using minao='sto-3g' if IAO convergence fails."
            )
            logger.info("Heavy TM detected (all-electron), using minao='%s'", effective)

        return effective, warnings
```

### 파일 6: `src/qcviz_mcp/renderers/pyvista_renderer.py` (신규)

```python
# -*- coding: utf-8 -*-
"""PyVista 기반 네이티브 오비탈 렌더러.

Playwright+Chromium (~200MB) 대신 PyVista+vtk-osmesa (~50MB)를 사용하여
브라우저 없이 직접 큐브 데이터 → PNG를 생성한다.

사용 파이프라인:
    cube_text = backend.generate_cube(mol, coeff, idx)
    parsed = parse_cube_string(cube_text)
    render_orbital_png(parsed["data"], parsed["origin"], parsed["axes"],
                       parsed["npts"], output_path="orbital.png")

설치:
    pip install pyvista
    # Linux 헤드리스: pip install vtk-osmesa (기존 vtk 대체, ~50MB)
"""
from __future__ import annotations

import logging
from pathlib import Path
from typing import Optional

import numpy as np

logger = logging.getLogger(__name__)

try:
    import pyvista as pv
    _HAS_PYVISTA = True
except ImportError:
    _HAS_PYVISTA = False


def is_available() -> bool:
    """PyVista 사용 가능 여부."""
    return _HAS_PYVISTA


def cube_to_pyvista_grid(
    cube_data: np.ndarray,
    origin: tuple[float, float, float],
    axes: list[np.ndarray],
    npts: tuple[int, int, int],
) -> "pv.ImageData":
    """큐브 데이터를 PyVista ImageData로 변환.

    Args:
        cube_data: 3D numpy array (nx, ny, nz).
        origin: 큐브 원점 (x, y, z) in Bohr.
        axes: 3개 축 벡터 리스트, 각각 전체 범위 벡터 (3,).
        npts: 각 축의 그리드 포인트 수 (nx, ny, nz).

    Returns:
        pv.ImageData
    """
    if not _HAS_PYVISTA:
        raise ImportError("PyVista not installed. Run: pip install pyvista")

    spacing = tuple(
        float(np.linalg.norm(ax) / max(n - 1, 1))
        for ax, n in zip(axes, npts)
    )

    grid = pv.ImageData(
        dimensions=npts,
        spacing=spacing,
        origin=origin,
    )
    grid["orbital"] = cube_data.flatten(order="F")
    return grid


def render_orbital_png(
    cube_data: np.ndarray,
    origin: tuple[float, float, float],
    axes: list[np.ndarray],
    npts: tuple[int, int, int],
    output_path: str = "orbital.png",
    isovalue: float = 0.02,
    window_size: tuple[int, int] = (1920, 1080),
    colors: tuple[str, str] = ("blue", "red"),
    background: str = "white",
    show_atoms: Optional[list[tuple[str, list[float]]]] = None,
) -> str:
    """큐브 데이터로부터 아이소서피스 PNG를 직접 생성.

    Args:
        cube_data: 3D 오비탈 데이터.
        origin, axes, npts: 큐브 파라미터.
        output_path: 출력 PNG 파일 경로.
        isovalue: 아이소서피스 값 (양의 로브).
        window_size: (width, height).
        colors: (양의 로브 색, 음의 로브 색).
        background: 배경색.
        show_atoms: [(symbol, [x,y,z]), ...] 원자 오버레이.

    Returns:
        str: 생성된 PNG 파일 경로.
    """
    if not _HAS_PYVISTA:
        raise ImportError("PyVista not installed. Run: pip install pyvista")

    pv.OFF_SCREEN = True

    grid = cube_to_pyvista_grid(cube_data, origin, axes, npts)
    pl = pv.Plotter(off_screen=True, window_size=window_size)
    pl.background_color = background

    # 양의 로브
    try:
        pos = grid.contour([isovalue], scalars="orbital")
        if pos.n_points > 0:
            pl.add_mesh(pos, color=colors[0], opacity=0.6, smooth_shading=True)
    except Exception as e:
        logger.debug("Positive contour failed: %s", e)

    # 음의 로브
    try:
        neg = grid.contour([-isovalue], scalars="orbital")
        if neg.n_points > 0:
            pl.add_mesh(neg, color=colors[1], opacity=0.6, smooth_shading=True)
    except Exception as e:
        logger.debug("Negative contour failed: %s", e)

    # 원자 위치 오버레이
    if show_atoms is not None:
        _ATOM_COLORS = {
            "H": "white", "C": "gray", "N": "blue", "O": "red",
            "F": "green", "S": "yellow", "Fe": "orange",
            "Ti": "silver", "Zr": "teal", "Mo": "purple",
        }
        for symbol, coord in show_atoms:
            sphere = pv.Sphere(radius=0.3, center=coord)
            pl.add_mesh(sphere, color=_ATOM_COLORS.get(symbol, "gray"), opacity=1.0)

    pl.camera_position = "iso"
    pl.camera.zoom(1.3)
    Path(output_path).parent.mkdir(parents=True, exist_ok=True)
    pl.screenshot(output_path)
    pl.close()

    file_size = Path(output_path).stat().st_size
    logger.info("PyVista PNG: %s (%d bytes)", output_path, file_size)
    return output_path


def render_from_cube_string(
    cube_text: str,
    output_path: str = "orbital.png",
    isovalue: float = 0.02,
    window_size: tuple[int, int] = (1920, 1080),
    colors: tuple[str, str] = ("blue", "red"),
    background: str = "white",
) -> str:
    """Gaussian cube 문자열 → PNG 원스텝 변환.

    기존 generate_cube() 출력을 바로 받아 PNG로 렌더링.
    parse_cube_string → render_orbital_png 파이프라인을 하나로 묶음.

    Args:
        cube_text: Gaussian cube format 문자열 (generate_cube() 반환값).
        output_path: 출력 PNG 경로.
        isovalue, window_size, colors, background: 렌더링 파라미터.

    Returns:
        str: 생성된 PNG 파일 경로.
    """
    from qcviz_mcp.backends.pyscf_backend import parse_cube_string

    parsed = parse_cube_string(cube_text)

    # 원자 위치를 오버레이용으로 변환
    _Z_TO_SYM = {
        1: "H", 6: "C", 7: "N", 8: "O", 9: "F", 16: "S",
        22: "Ti", 26: "Fe", 40: "Zr", 42: "Mo",
    }
    show_atoms = [
        (_Z_TO_SYM.get(z, "X"), [x, y, z_coord])
        for z, x, y, z_coord in parsed["atoms"]
    ]

    return render_orbital_png(
        parsed["data"], parsed["origin"], parsed["axes"], parsed["npts"],
```

```python
def verify_basis_independence(
    molecule_name: str,
    results_by_basis: dict[str, dict],
) -> dict:
    """여러 기저의 IBO 결과를 비교하여 기저 독립성 검증.

    Knizia 2013 원논문에 따르면, IAO 투영 덕분에 IBO 수와
    화학적 해석이 기저 함수에 독립적이어야 한다.

    Args:
        molecule_name: 분자 이름 (로깅용).
        results_by_basis: {basis_name: {"n_ibo": int, "charges": np.ndarray}}

    검증 기준:
    - IBO 수: 모든 기저에서 동일
    - 전하 합: |Σq| < 1e-4
    - 원자별 전하 편차: 기저 간 max |Δq| < 0.15 e

    Returns:
        dict: ibo_count_invariant, charge_conservation,
              charge_deviation_ok, max_charge_deviation,
              ibo_counts, all_passed.
    """
    if len(results_by_basis) < 2:
        return {
            "ibo_count_invariant": True,
            "charge_conservation": True,
            "charge_deviation_ok": True,
            "max_charge_deviation": 0.0,
            "ibo_counts": {b: r["n_ibo"] for b, r in results_by_basis.items()},
            "all_passed": True,
        }

    ibo_counts = {b: r["n_ibo"] for b, r in results_by_basis.items()}
    ibo_invariant = len(set(ibo_counts.values())) == 1

    charge_conservation = True
    for b, r in results_by_basis.items():
        if "charges" in r and r["charges"] is not None:
            if abs(float(np.sum(r["charges"]))) >= 1e-4:
                charge_conservation = False

    charge_arrays = [
        r["charges"] for r in results_by_basis.values()
        if "charges" in r and r["charges"] is not None
    ]

    max_deviation = 0.0
    if len(charge_arrays) >= 2:
        for i in range(len(charge_arrays)):
            for j in range(i + 1, len(charge_arrays)):
                if len(charge_arrays[i]) == len(charge_arrays[j]):
                    dev = float(np.max(np.abs(charge_arrays[i] - charge_arrays[j])))
                    max_deviation = max(max_deviation, dev)

    charge_deviation_ok = max_deviation < 0.15
    all_passed = ibo_invariant and charge_conservation and charge_deviation_ok

    if not all_passed:
        logger.warning("%s basis independence FAILED: IBO=%s, maxΔq=%.4f",
                       molecule_name, ibo_counts, max_deviation)
    else:
        logger.info("%s basis independence PASSED (maxΔq=%.4f)",
                    molecule_name, max_deviation)

    return {
        "ibo_count_invariant": ibo_invariant,
        "charge_conservation": charge_conservation,
        "charge_deviation_ok": charge_deviation_ok,
        "max_charge_deviation": max_deviation,
        "ibo_counts": ibo_counts,
        "all_passed": all_passed,
    }
```

```python
        for basis in EXTENDED_BASIS_SETS:

            try:
                scf_res, mol_obj = backend.compute_scf(mol.atom_spec, basis=basis, method="HF")
                iao_res = backend.compute_iao(scf_res, mol_obj)

                      f"{elapsed:.1f}s {status}")

                mol_results[basis] = {
                    "n_ibo": ibo_res.n_ibo,
                    "charges": iao_res.charges,
                    "energy": scf_res.energy_hartree,
```

```markdown
## [0.6.0-alpha] - 2026-03-05

### Added

- SFX2C-1e scalar relativistic correction for 4d/5d transition metals
  - `compute_scf_relativistic()` with SFX2C-1e Hamiltonian
  - Tested on ZrCl₄ (4d) and Mo(CO)₆ (4d)
- Adaptive SCF convergence escalation engine (`ConvergenceStrategy`)
  - 5-level auto-escalation: DIIS → level_shift → damp → SOSCF → SOSCF+shift
  - `ConvergenceError` for total failure; immutable LEVELS (tuple, not list)
  - `compute_scf_adaptive()` and `compute_scf_flexible(adaptive=True)`
- PyVista native orbital renderer (`renderers/pyvista_renderer.py`)
  - Direct cube data → PNG without browser (~50MB vs Playwright ~200MB)
  - `render_orbital_png()`, `render_from_cube_string()`
  - Automatic renderer selection: PyVista > Playwright > HTML-only
- `parse_cube_string()` — Gaussian cube text → numpy 3D array parser
  - Bridges `generate_cube()` output to PyVista renderer
- Extended basis set benchmarks: 10 molecules × 4 bases = 40 cases
  - `EXTENDED_BASIS_SETS`: sto-3g, cc-pvdz, cc-pvtz, aug-cc-pvdz
  - `run_extended_benchmark.py` with timing + independence report
- IBO basis-set independence verification (`verify_basis_independence()`)
- ZrCl₄ and Mo(CO)₆ heavy transition metal benchmark molecules
- `viz-native` optional dependency group (pyvista>=0.43)
- pytest custom markers: `slow`, `network`

### Changed

- `compute_scf_flexible()` accepts `adaptive=True` parameter
- `_resolve_minao()` warns on 4d/5d TM without ECP
- `ConvergenceStrategy.LEVELS` is a tuple (immutable, test-safe)
- `renderers/__init__.py` now includes auto-selection logic

### Design decisions vs original plan

- Lanthanide (Ce, Gd) → 4d TM (Zr, Mo): LANL2DZ ECP only covers La, not Ce-Lu
- `pyscf_backend.py`: additive patches, not full file replacement
- `parse_cube_string()` added to bridge generate_cube() → PyVista pipeline
```
